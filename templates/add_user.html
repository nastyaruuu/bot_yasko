Для того чтобы воспользоваться API и создать нового волонтера:



    import requests

data = {

  "login": "логин",

  "password": "пароль",

  "surname": "фамилия",

  "name": "имя",

  "age": "11",

  "speciality": "специальность",

  "adress": "адрес"

}

url = 'http://my-domains.ru/api/volunteers/add'

response = requests.get(url, json=data)

if response.status_code == 200:

    print("Запрос успешно отправлен!")

else:

    print("Произошла ошибка при отправке запроса.")
