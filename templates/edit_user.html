Для того чтобы воспользоваться API и редактировать волонтера через API: 


import requests


    data = {

      "login": "логин",

      "password": "пароль",

      "surname": "фамилия",

      "name": "имя",

      "age": "11",

      "speciality": "специальность",

      "adress": "адрес"

    }

    id = '1'

    url = 'http://my-domains.ru/api/volunteers/edit'

    response = requests.put(url+id, json=data)

    
    if response.status_code == 200:

        print("Запрос успешно отправлен!")

    else:

        print("Произошла ошибка при отправке запроса.")
